CREATE DATABASE `musica` /*!40100 DEFAULT CHARACTER SET utf8 */;


CREATE TABLE `musica.artista` (
  `CODIGO_ARTISTA` int(11) NOT NULL AUTO_INCREMENT,
  `NOME` varchar(100) NOT NULL,
  `NOME_ARTISTICO` varchar(100) NOT NULL,
  PRIMARY KEY (`CODIGO_ARTISTA`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

  
CREATE TABLE `musica.musica` (
  `CODIGO_MUSICA` int(11) NOT NULL AUTO_INCREMENT,
  `NOME` varchar(100) NOT NULL,
  `GENERO` varchar(100) NOT NULL,
  `DURACAO` varchar(100) NOT NULL,
  `CODIGO_ARTISTA` int(11) DEFAULT NULL,
  PRIMARY KEY (`CODIGO_MUSICA`),
  KEY `FK_ARTISTA_idx` (`CODIGO_ARTISTA`),
  CONSTRAINT `FK_ARTISTA` FOREIGN KEY (`CODIGO_ARTISTA`) REFERENCES `artista` (`CODIGO_ARTISTA`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=10 DEFAULT CHARSET=utf8;
